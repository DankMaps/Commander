Server Reboot Guide

1. Stop Services

systemctl stop b1s
systemctl stop sapb1servertools

2. Switch to Relevant Users

su seiadm
su ndbadm 
su hdbadm
su texadm

3. Stop HDB

csh 
./HDB stop
exit
exit

4. Stop SAP Initialization

sudo /etc/init.d/sapinit stop
systemctl stop sapinit

5. Reboot the System

reboot

6. Start SAP Initialization

systemctl start sapinit

7. Switch to Relevant Users

su seiadm
su ndbadm 
su hdbadm
su texadm

8. Start HDB

csh 
./HDB start
exit
exit

9. Restart Services

systemctl stop b1s
systemctl stop sapb1servertools
systemctl start sapb1servertools
systemctl start b1s
